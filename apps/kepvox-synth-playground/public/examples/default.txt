import { OscillatorSynthNode, Unit } from 'synth'
import { context } from 'synth/playground'

const oscillator = new OscillatorSynthNode(context)

oscillator.connectOutput(context.output)

oscillator.frequency.initialValue = Unit.hertz.orThrow(440)
oscillator.frequency.curve.holdValueAt(context.time({ note2: 1 }))
oscillator.frequency.curve.rampValueUntil(context.time({ note2: 1, note4: 2 }), Unit.hertz.orThrow(880))
